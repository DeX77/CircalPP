48  void ScoringModel::read(std::istream & input)
49  {
50  if (!input)
51  {
52  throw bpp::IOException(
53  "ScoringModel::ScoringModel: fail to open file");

56  std::string temp;
57  std::string type;

59  double gapStart = 0;

61  double match = 0;
62  double missmatch = 0;
63  double negativeMatch = 0;
64 
65  double maxValue = 1;
66 
67  uint Size = 1;
68 
69  // Main loop : for all file lines
70  while (!input.eof())
71  {
72  getline(input, temp, '\n'); // Copy current line in temporary string
73 
74  if (temp[0] != '#')
75  {
76  std::istringstream istring(temp);
77  istring >> type >> gapStart >> gapExtend >> match
78  >> negativeMatch >> missmatch;
79  istring >> temp; //Cut out the ":" char
80 
81  if (match > maxValue)

83 
84  if (negativeMatch > maxValue)
85  maxValue = negativeMatch;
86 
87  while (istring)
88  {
89  AlignmentSymbol sc;
90  istring >> temp;
91  if (!temp.empty())
92  {
93  sc.symbol = bpp::TextTools::toUpper(temp);
94  if (sc.symbol.length() > Size)
95  Size = sc.symbol.length();
96  sc.type = type;
97  //Changed gapOpen & gapExtend to negative Values to correspond with Maximize
98  sc.gapOpen = -gapStart;
99  sc.gapExtend = -gapExtend;
100  sc.match = match;
101  sc.negativeMatch = negativeMatch;
102  sc.missmatch = -missmatch;
103 
104  model.insert(

106  sc.symbol, sc));
107  //Also add negative Symbol Version
108  sc.symbol = "-" + sc.symbol;
109  model.insert(

111  sc.symbol, sc));

128  double ScoringModel::ScoreOf(const std::string &A, const std::string &B)
129  {
130 
131  double outA = 0;
132  double outB = 0;
133 
134  //Check if two symbols are completly identical
135  if (A.compare(B) == 0)
136  {
137  return model[A].match;
138  }
139  else
140  //Check for negative values
141  if ((A.compare("-" + B) == 0) || (B.compare("-" + A) == 0))
142  {
143  return model[A].negativeMatch;
144  }
145  else
146  {
147  outA = model[A].missmatch;
148  outB = model[B].missmatch;
149  //Check wich one is worse
150  if (BestOfTwo(outA, outB) != outA)
151  return outA;
152  else
153  return outB;
154  }
